69def forward(baseUrl,endpoint, method, data=None):
70 url = f"{baseUrl}{endpoint}"
71 try:
72 if method == 'GET':
73 response = requests.get(url)
74 elif method == 'POST':
75 response = requests.post(url, json=data)
76 elif method == 'PUT':
77 response = requests.put(url, json=data)
78 elif method == 'DELETE':
79 response = requests.delete(url)
80 else:
81 return jsonify({"error": "Unsupported method"}), 405
82
83 response.raise_for_status()
84 return jsonify(response.json()), response.status_code
85 except requests.exceptions.RequestException as e:
86 return jsonify({"error": str(e)}), 500
