76@app.route('/control/<int:gnbId>/<string:command>', methods=['POST'])
77def gnb_control_start(command, gnbId):
78 """
79 @brief Sends start/stop command to NT-RIC, O-CU, O-DU, and O-RU.
80 @param command The command to execute (start/stop).
81 @param gnbId The ID of the gNB.
82 @return JSON response with the result of the command.
83 """
84 if command not in ['start', 'stop']:
85 return bad_request("Invalid command")
86 cgnbcf_data_store.setdefault(gnbId, {})['gnb_control'] = command
87 return jsonify({"result": f"gNB control {command} command executed successfully"}), 201
